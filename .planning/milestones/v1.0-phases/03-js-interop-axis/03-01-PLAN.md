Interop facts (RESEARCH Pattern 2 + Code Examples):
- [<Emit("$0 + $1")>] on `let add (x:int) (y:int):int = jsNative` → compiles the call site to `(x + y)` inline; $0=first arg, $1=second. jsNative is the placeholder body.
- [<Import("greet", "./helpers.js")>] on `let greet (name:string):string = jsNative` → emits `import { greet } from "./helpers.js"`. Function form `importMember "greet" "./helpers.js"` (needs `open Fable.Core.JsInterop`) is equivalent.
- importDefault "./helpers.js" → emits `import helpers from "./helpers.js"` (default export).
- [<Global("Math")>] on `let mathObj : obj = jsNative` → binds the existing global `Math` with NO import statement; `mathObj?random()` → `Math.random()`. A function-typed [<Global>] also works (Ch.3 used `[<Global>] let consoleLog`).
- Dynamic `?` (needs `open Fable.Core.JsInterop`): `obj?prop` → property access; `obj?method(args)` → method call. Produces untyped obj — teach as escape hatch, not production default.
- RESEARCH Pitfall 2 (import lookup table): JS `import x from 'lib'` → F# `importDefault`; JS `import { y } from 'lib'` → F# `import "y" "lib"`; JS `import * as ns` → F# `importAll` (rare). Teach this table.

module App

// ANCHOR: emit-example

// ANCHOR_END: global-example

// ANCHOR: import-example
// [<Import("name","module")>] emits: import { greet } from "./helpers.js"
[<Import("greet", "./helpers.js")>]
let greet (name: string) : string = jsNative
// importDefault emits: import helpers from "./helpers.js"
let helpers : obj = importDefault "./helpers.js"
// ANCHOR_END: import-example

// ANCHOR: dynamic-example
// The ? operator (needs open Fable.Core.JsInterop) gives untyped dynamic access.
let shouted : string = helpers?shout("fable")   // → helpers.shout("fable")
let libVersion : string = helpers?version        // → helpers.version
// ANCHOR_END: dynamic-example

let app = document.getElementById "app"
app.innerHTML <-
    sprintf "<h1>기본 Interop</h1><p>jsAdd 3 4 = %d</p><p>%s</p><p>shout: %s (v%s)</p><p>random: %f</p>"
